cmake_minimum_required(VERSION 3.22.1)

project(mystery)

set(CMAKE_CXX_STANDARD 17)
set(CMAKE_CXX_STANDARD_REQUIRED ON)
set(CMAKE_CXX_EXTENSIONS OFF)

add_library(mystery SHARED
    mystery.cpp
)

target_link_options(mystery PRIVATE
    "-Wl,-z,max-page-size=16384"
    "-Wl,-z,common-page-size=16384"
)

set_target_properties(mystery PROPERTIES
    SUFFIX ".so"
    PREFIX "lib"
    VERSION 1.0.0
    SOVERSION 1
)

if(ANDROID_ABI STREQUAL "x86")
    target_compile_options(mystery PRIVATE
        -ffast-math
        -mtune=atom
        -mssse3
        -mfpmath=sse
        -O3
    )
elseif(ANDROID_ABI STREQUAL "x86_64")
    target_compile_options(mystery PRIVATE
        -ffast-math
        -mtune=generic
        -O3
    )
elseif(ANDROID_ABI STREQUAL "armeabi-v7a")
    target_compile_options(mystery PRIVATE
        -ffast-math
        -mtune=cortex-a8
        -mfpu=neon
        -O3
    )
elseif(ANDROID_ABI STREQUAL "arm64-v8a")
    target_compile_options(mystery PRIVATE
        -ffast-math
        -O3
    )
endif()

find_library(log-lib log)
find_library(android-lib android)

target_link_libraries(mystery
    ${log-lib}
    ${android-lib}
)

target_compile_definitions(mystery PRIVATE
    MYSTERY_VERSION=1
)
